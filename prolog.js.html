var artemis;
(function (artemis) {
    /**
     * Gets Class Metadata - Name
     *
     * @param klass
     * @returns {string|SVGAnimatedString|string|string|string|string|*}
     */
    function getClassName(klass) {
        return klass.className || klass.name;
    }
    artemis.getClassName = getClassName;
})(artemis || (artemis = {}));
